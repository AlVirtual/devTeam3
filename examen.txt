function llencaError($param){
    if (!isset($param)){
        throw new Exception('There is an error.');
    } else {
        return $param;
    }
}

function calculadora($v1, $v2, $operacio){
    switch($operacio){
        case("suma"):
            return $v1+$v2;
            break;
        case("resta"):
            return $v1-$v2;
            break;
        case("multiplicació"):
            return $v1*$v2;
            break;
    }
}

function tornaElement($valors, $posicio){
    return $valors[$posicio-1];
}
